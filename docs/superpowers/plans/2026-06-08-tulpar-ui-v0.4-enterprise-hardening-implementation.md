/**
 * Dev-only console.warn. In production bundles, the conditional is
 * tree-shaken by Vite/Rollup constant folding around `import.meta.env.DEV`
 * and `process.env.NODE_ENV`.
 */
export function warnDev(msg: string, ctx?: unknown): void {
  const isDev =
    (typeof import.meta !== "undefined" &&
      (import.meta as unknown as { env?: { DEV?: boolean } }).env?.DEV === true) ||
    (typeof process !== "undefined" && process.env?.NODE_ENV !== "production");
  if (isDev) {
    if (ctx === undefined) console.warn(msg);
    else console.warn(msg, ctx);
  }
}
```

import { writeFile } from "node:fs/promises";
import { dirname, resolve } from "node:path";
import { fileURLToPath } from "node:url";
import { primitiveColor } from "../src/primitive/color";

export interface FallbackValues {
  borderDefault: string;
  textPrimary: string;
  textInverse: string;
  bgSurface: string;
}

export function buildFallbacks(): FallbackValues {
  return {
    borderDefault: primitiveColor.stone[200],
    textPrimary: primitiveColor.stone[900],
    textInverse: primitiveColor.stone[50],
    bgSurface: primitiveColor.white,
  };
}

function emit(fb: FallbackValues): string {
  return `/* AUTO-GENERATED by build/generate-fallbacks.ts — do not edit by hand. */
export const FALLBACKS = ${JSON.stringify(fb, null, 2)} as const;

export type FallbackKey = keyof typeof FALLBACKS;
`;
}

async function main() {
  const __dirname = dirname(fileURLToPath(import.meta.url));
  const out = resolve(__dirname, "../src/fallbacks.ts");
  await writeFile(out, emit(buildFallbacks()));
  console.log("✅ Generated fallbacks:", out);
}

if (import.meta.url === `file://${process.argv[1]}` || process.argv[1]?.endsWith("generate-fallbacks.ts")) {
  main().catch((err) => {
    console.error(err);
    process.exit(1);
  });
}
```

private _handleClick = (e: Event): void => {
  if (this.disabled || this.loading) {
    e.preventDefault();
    e.stopImmediatePropagation();
    return;
  }
  if (this.href) return; // anchor handles its own activation
  const form = this._internals.form;
  if (!form) return;
  if (this.type === "submit") {
    e.preventDefault();
    form.requestSubmit(this);
  } else if (this.type === "reset") {
    e.preventDefault();
    form.reset();
  }
};

# Migrating from Tulpar UI v0.3.x to v0.4

v0.4 is a hardening release: 6 bug fixes and 14 enterprise-grade improvements.
Because Tulpar UI is pre-1.0, this release ships behavior changes alongside
the additive improvements. This guide walks every observable change.

## Peer dependency bumps

| Package | Old | New |
|---|---|---|
| `@angular/core` | `^20.0.0` | `^22.0.0` |
| `@angular/common` | `^20.0.0` | `^22.0.0` |
| `vue` | `^3.4.0` | `^3.5.0` |
| `lucide-angular` | `>=0.400.0` | `>=0.500.0` |
| `zone.js` | (dev) | removed — Angular 22 zoneless default |

Browser baseline: Chromium 65+, Firefox 78+, **Safari 17+**.

## Behavior changes

### Form submission

`<tulpar-button type="submit">` no longer dispatches duplicate submit events.
If your app relied on the old behavior, update your handler to expect exactly
one submit per click.

`name` and `value` are now reflected to the parent `<form>` on submission,
matching native `<button>`. Apps that added hidden inputs to work around this
should remove those workarounds (otherwise FormData will contain duplicate
keys).

### Disabled anchor

`<tulpar-button href="…" disabled>` no longer sets `href="javascript:void(0)"`.
The anchor remains inert via `aria-disabled="true"` and `tabindex="-1"`. CSP
configurations that blocked `javascript:` URLs no longer trigger violations.

### ButtonGroup ARIA role

`<tulpar-button-group>` host now exposes `role="toolbar"` (was `group`) with
`aria-orientation="horizontal"` (or `vertical` when stacked). Screen reader
users hear "toolbar" instead of "group".

Apps that queried `[role="group"]` to find button groups must update to
`[role="toolbar"]`. Note: the role is set via `ElementInternals.role`, not as
a DOM attribute, so `getAttribute("role")` returns `null` — query by element
tag (`tulpar-button-group`) or by accessible name.

### Focus ring

Focus indication switched from `box-shadow` to `outline`. Visual position is
similar (2px outside the button). Apps that disabled the focus ring via
`box-shadow: none` overrides should use `outline: none` instead.

The ring now responds to `forced-colors: active` (Windows High Contrast Mode),
where it adopts `CanvasText` for system contrast.

### Console output

Development-mode warnings unchanged. Production bundles no longer log button
ARIA validation warnings (a11y checks happen in dev only).

### Wrapper DOM (Angular)

`<tulpar-button-ng>` host now uses `display: contents`. The wrapper element
is excluded from layout, becoming transparent to the box tree. CSS that
targets `tulpar-button-ng { ... }` for layout (width, padding, border) will
not take effect — apply layout to `<tulpar-button>` or wrap externally.

### Safari 16

`delegatesFocus` on form-associated custom elements is unreliable in Safari
≤ 16. Tulpar UI v0.4 baselines Safari 17+. On Safari 16, focus still reaches
the inner button, but the focus ring on that inner button may not render.

## Type imports

The wrapper packages no longer redefine the type unions. Consumers can keep
importing from the wrapper packages (they re-export from core) or switch to:
```ts
import type { ButtonSeverity, ButtonVariant } from "@tulpar-ui/core";
```

## New props on `tulpar-button`

| Prop | Type | Purpose |
|---|---|---|
| `name` | string | Form submission key (matches native `<button>`) |
| `value` | string | Form submission value (matches native `<button>`) |
```
